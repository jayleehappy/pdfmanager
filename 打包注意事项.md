# PyInstaller 打包注意事项

> 本文档记录 PDF 名册管理生成工具（pdfmc）打包过程中遇到的所有问题及解决方案，
> 供后续维护和打包参考。
> 适用环境：Windows + Python 3.11 + PyInstaller 6.x

---

## 一、核心结论（先读这条）

**使用 onedir 模式（目录分发），不要使用 onefile 模式。**

原因：
- onefile 在运行时将内容解压到 `Temp\_MEIxxxxxx\` 目录，Windows Shell COM 子系统初始化不完整
- 导致 `QFileDialog.getExistingDirectory`（目录选择）静默失败，但 `getOpenFileName`（文件选择）正常
- onedir 模式下 exe 和所有依赖在同级目录，稳定性高得多

## 二、spec 文件关键配置

### 2.1 完整 spec 模板

```python
# -*- mode: python ; coding: utf-8 -*-
import sys, os
from pathlib import Path

block_cipher = None

# 项目根目录
_spec_path = getattr(sys, 'frozen', False)
if _spec_path:
    base_dir = Path(getattr(sys, '_MEIPASS', _spec_path)).parent
else:
    base_dir = Path(sys.argv[0]).parent.resolve()

# ============ 依赖路径（开发环境下解析）============
try:
    import PySide6
    pyside6_plugins_src = (Path(PySide6.__file__).parent / "plugins").resolve()
except Exception:
    pyside6_plugins_src = None

try:
    import pypdfium2
    import pypdfium2_raw as _raw
    pypdfium2_pkg_dir = Path(pypdfium2.__file__).parent.resolve()
    pypdfium2_raw_pkg_dir = Path(_raw.__file__).parent.resolve()
    del _raw
except Exception:
    pypdfium2_pkg_dir = None
    pypdfium2_raw_pkg_dir = None

_python_dir = Path(sys.executable).parent

# ============ datas（Python 包、资源文件）============
datas = [
    (str(base_dir / "config.py"), "."),
    (str(base_dir / "src"), "src"),
]

# pypdfium2 完整包（含 Python 文件）
if pypdfium2_pkg_dir and pypdfium2_pkg_dir.exists():
    datas.append((str(pypdfium2_pkg_dir), "pypdfium2"))

# pypdfium2_raw 完整包（含 pdfium.dll + bindings.py）
# ⚠️ pypdfium2_raw 是 site-packages 顶级包，不是 pypdfium2 的子包
# 导入链：pypdfium2.raw → from pypdfium2_raw.bindings import *
if pypdfium2_raw_pkg_dir and pypdfium2_raw_pkg_dir.exists():
    datas.append((str(pypdfium2_raw_pkg_dir), "pypdfium2_raw"))

# PySide6 Qt 插件
if pyside6_plugins_src and pyside6_plugins_src.exists():
    for plugin_dir in ["platforms", "imageformats", "iconengines"]:
        p = pyside6_plugins_src / plugin_dir
        if p.exists():
            datas.append((str(p), f"PySide6/plugins/{plugin_dir}"))

# ============ binaries（DLL 文件）============
binaries = []

# ⚠️ vcruntime140_1.dll 不在 C:\Windows\System32，必须显式打包！
for _dll in ("vcruntime140.dll", "vcruntime140_1.dll"):
    _dll_path = _python_dir / _dll
    if _dll_path.exists():
        binaries.append((str(_dll_path), "."))

# Windows COM 核心库（SHBrowseForFolder 需要）
_win32_dlls = {
    "C:/Windows/System32/ole32.dll": ".",
    "C:/Windows/System32/shell32.dll": ".",
}
for _dll_path, _dest in _win32_dlls.items():
    if Path(_dll_path).exists():
        binaries.append((_dll_path, _dest))

# ============ Analysis ================
a = Analysis(
    [str(base_dir / "main.py")],
    pathex=[str(base_dir)],
    binaries=binaries,
    datas=datas,
    hiddenimports=[
        "pypdf", "pypdfium2",
        "PIL", "PIL.Image",         # ⚠️ 勿从 excludes 中排除 PIL
        "PySide6", "PySide6.QtCore", "PySide6.QtGui",
        "PySide6.QtWidgets", "PySide6.QtPrintSupport",
        "shiboken6",
        "xml.etree.ElementTree", "xml.dom.minidom",
        "pathlib", "packaging.version",
    ],
    runtime_hooks=[str(base_dir / "rthook.py")],
    excludes=[
        "matplotlib", "numpy", "pandas", "scipy",
        # ⚠️ 不要排除 PIL，否则 page.render().to_pil() 失败
        "tkinter", "test", "unittest",
    ],
    win_no_prefer_redirects=False,
    win_private_assemblies=False,
    cipher=block_cipher,
    noarchive=False,
)

pyz = PYZ(a.pure, a.zipped_data, cipher=block_cipher)

exe = EXE(
    pyz, a.scripts, [],
    exclude_binaries=True,
    name="pdfmc",
    debug=False,
    bootloader_ignore_signals=False,
    strip=False, upx=False,
    console=False,               # GUI 程序关闭控制台
    icon=None,
)

# ============ Onedir（目录分发）============
coll = COLLECT(
    exe, a.binaries, a.zipfiles, a.datas,
    strip=False, upx=False,
    outdir=str(base_dir / "dist"),
    name="pdfmc",
)
```

### 2.2 hiddenimports 关键项说明

| 包 | 必须加入 hiddenimports 的原因 |
|----|------|
| `pypdfium2` | PyInstaller 无法自动识别动态导入 |
| `PIL` / `PIL.Image` | `page.render().to_pil()` 依赖 Pillow |
| `shiboken6` | PySide6 核心，必须显式声明 |
| `packaging.version` | PyInstaller 自身依赖 |

### 2.3 excludes 注意事项

- **不要排除** `PIL`：pdfium 渲染需要 Pillow 将结果转为 PIL Image
- **不要排除** `tkinter`：虽然本程序不用，但 PyInstaller 自身可能依赖

## 三、rthook.py 运行时钩子

在程序启动最早阶段执行，必须处理三件事：

### 3.1 强制 sys.path 优先级（解决包导入冲突）

```python
# -*- coding: utf-8 -*-
import sys, os

# ⚠️ 这一段必须放在最开头
if getattr(sys, 'frozen', False) and hasattr(sys, '_MEIPASS'):
    _internal = sys._MEIPASS
    while _internal in sys.path:
        sys.path.remove(_internal)
    sys.path.insert(0, _internal)  # 确保 bundled 包优先于系统 site-packages
```

**为什么需要**：PyInstaller 引导程序将 `_internal` 加到 `sys.path` 时位置不够靠前，
Python 会先找到系统 site-packages 中的同名包（缺文件/不完整），导致导入失败。

### 3.2 Windows DLL 搜索路径

```python
if sys.platform == "win32" and hasattr(os, "add_dll_directory"):
    candidates = []
    exe_dir = os.path.dirname(sys.executable)
    candidates.append(exe_dir)
    if getattr(sys, 'frozen', False):
        meipass = getattr(sys, '_MEIPASS', '')
        if meipass and meipass != exe_dir:
            candidates.append(meipass)
    for p in sys.path:
        if p and os.path.isfile(os.path.join(p, "python311.dll")):
            candidates.append(p)
    added = set()
    for d in candidates:
        if d and d not in added and os.path.isdir(d):
            try:
                os.add_dll_directory(d)
                added.add(d)
            except Exception:
                pass
```

### 3.3 Windows COM 初始化（目录选择对话框必需）

```python
if sys.platform == "win32":
    try:
        import ctypes
        ole32 = ctypes.WinDLL("ole32")
        ole32.OleInitialize(None)  # ⚠️ 初始化 COM，否则 SHBrowseForFolder 失败
    except Exception:
        pass
```

### 3.4 Qt 插件路径（onedir vs onefile 路径不同）

```python
if getattr(sys, 'frozen', False) and hasattr(sys, '_MEIPASS'):
    meipass = sys._MEIPASS
    if os.path.basename(meipass) == "_internal":
        # onedir: _MEIPASS = _internal，exe_dir = 其父目录
        exe_dir = os.path.dirname(meipass)
    else:
        # onefile: _MEIPASS = 临时解压目录
        exe_dir = os.path.dirname(sys.executable)
else:
    exe_dir = os.path.dirname(os.path.abspath(__file__))

_internal = os.path.join(exe_dir, "_internal")
plugin_path = os.path.join(_internal, "PySide6", "plugins")
if os.path.isdir(plugin_path):
    os.environ['QT_PLUGIN_PATH'] = plugin_path
    try:
        from PySide6 import QtCore
        QtCore.QCoreApplication.setLibraryPaths([plugin_path])
    except Exception:
        pass
```

## 四、pypdfium2 打包的坑

### 4.1 包结构（开发环境 site-packages）

```
site-packages/
├── pypdfium2/                    ← 主包（Python 文件）
│   ├── __init__.py
│   ├── raw.py                    ← from pypdfium2_raw.bindings import *
│   ├── _helpers/
│   └── ...
└── pypdfium2_raw/               ← ⚠️ 顶级包，不是 pypdfium2 的子包
    ├── __init__.py
    ├── bindings.py               ← pypdfium2.raw 依赖此文件
    └── pdfium.dll               ← PDF 渲染引擎
```

### 4.2 常见错误汇总

| 报错信息 | 原因 | 解决 |
|---------|------|------|
| `No module named 'pypdfium2'` | `pypdfium2` 包未被打包 | 在 datas 中添加完整包 |
| `No module named 'pypdfium2_raw'` | `pypdfium2_raw` 未被打包 | 通过 `import pypdfium2_raw` 获取路径并打包 |
| `No module named 'PIL'` | PIL 被排除或未打包 | 从 excludes 移除 "PIL"，加入 hiddenimports |
| `ImportError: No module named 'bindings'` | `pypdfium2_raw` 完整包未打包 | 必须同时打包 `pypdfium2` 和 `pypdfium2_raw` |
| `OSError: cannot load library 'pdfium.dll'` | DLL 不在 `pypdfium2_raw/` 目录 | 确保 `pypdfium2_raw/` 包含 `pdfium.dll` |

### 4.3 验证打包是否正确

```python
# 解压后检查 _internal 目录结构
import os
_internal = "dist/pdfmc/_internal"
assert os.path.isdir(f"{_internal}/pypdfium2"), "pypdfium2 缺失"
assert os.path.isdir(f"{_internal}/pypdfium2_raw"), "pypdfium2_raw 缺失"
assert os.path.isfile(f"{_internal}/pypdfium2_raw/pdfium.dll"), "pdfium.dll 缺失"
assert os.path.isfile(f"{_internal}/pypdfium2_raw/bindings.py"), "bindings.py 缺失"
assert os.path.isdir(f"{_internal}/PIL"), "PIL 缺失"
```

## 五、Windows 系统 DLL 的坑

### 5.1 vcruntime140_1.dll 不在 System32

`vcruntime140_1.dll` 只存在于 Python 安装目录（`C:\Users\...\Python311\`），
不在 `C:\Windows\System32\`。PyInstaller 默认不会自动包含它。

**症状**：`failed to load python dll '...\python311.dll'`

**解决**：在 spec 的 `binaries` 中显式添加：
```python
for _dll in ("vcruntime140.dll", "vcruntime140_1.dll"):
    _dll_path = _python_dir / _dll
    if _dll_path.exists():
        binaries.append((str(_dll_path), "."))
```

### 5.2 ole32.dll / shell32.dll

`QFileDialog.getExistingDirectory`（原生模式）依赖 Windows Shell COM 组件：
- `SHBrowseForFolder` 需要 `ole32.dll`
- 部分 Shell 命名空间扩展需要 `shell32.dll`

这两个 DLL 在 System32 中，直接用路径添加即可。

## 六、onefile vs onedir 对比

| 特性 | onefile | onedir |
|------|---------|--------|
| 分发形式 | 单一 exe | exe + 文件夹 |
| 文件大小 | 较小（55MB） | 较大（157MB） |
| 运行稳定性 | ⚠️ Shell COM 易出问题 | ✅ 稳定 |
| 目录选择对话框 | ❌ 经常失败 | ✅ 正常 |
| 文件选择对话框 | ✅ 正常 | ✅ 正常 |
| PDF 预览 | ✅ | ✅ |
| 推荐用途 | 临时工具 | 正式发布 |

**结论**：正式发布使用 onedir，程序更稳定。

## 八、常用打包命令

```bash
# 清理并重新打包
rm -rf build dist
pyinstaller pdfmc.spec

# 打包时跳过已分析的模块（加速）
pyinstaller --conly pdfmc.spec    # 仅重新打包 exe
pyinstaller --distpath dist --workpath build pdfmc.spec  # 指定输出路径
```
